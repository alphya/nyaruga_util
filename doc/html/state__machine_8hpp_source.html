98  template <typename SharedData = detail::empty_data>
99  class state_machine
100  {
101  public:
102  using state_ptr = std::shared_ptr<state<SharedData>>;
103  using state_itr = typename std::map<std::string, state_ptr>::iterator;
104 
105  private:
106  SharedData m_data_;

109 
110  public:
111 
112  template<class T>
113  state_ptr add_state()
114  {
115  state_ptr state(new T, detail::my_deleter);
116  return add_state(state);
117  }
118 
119  state_ptr add_state(state_ptr state)
120  {
121  state->set_shared_data(&m_data_);
122  state->initialize();
123  state->change_state.connect(boost::bind(&state_machine::change_state, this, _1));
124  m_states_.insert(std::make_pair(state->get_label(), state));
125  return state;
126  }
127 
128  SharedData& getSharedData() noexcept
129  {
130  return m_data_;
131  }
132 
133  [[nodiscard]]
134  std::map<std::string, state_ptr> getStates() const noexcept
135  {
136  return m_states_;
137  }
138 
139  bool change_state(const std::string& label)
140  {
141  auto it = m_states_.find(label);
142 
143  if (it == m_states_.end()) return false;
144 
145  else if (it->second != m_current_state_)
146  {
147  if (m_current_state_) m_current_state_->state_exit();
148  m_current_state_ = it->second;
149  m_current_state_->state_enter();
150  return true;
151  }
152 
153  else return false;
154  }
155 
156  void update()
157  {
158  if (m_current_state_) m_current_state_->update();
159  }
