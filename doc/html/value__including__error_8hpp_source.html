33 // 偏微分^2 * 誤差^2 を、一変数分だけ計算して、再帰
34 template <std::size_t count, std::size_t max, typename F, typename... Args>
35 decltype(auto) generate_error_impl(F && f, Args &&... args)
36 {
37  if constexpr (count > max)
38  return num_t{ 0 };
39  else {
40  auto && current_val_obj = boost::hana::arg<count>(args...);
41  auto && current_val_error = current_val_obj.error;
42  auto && partial_diff = nyaruga::util::partial_diff<count>(f)(static_cast<num_t>(args)...);
43  auto && result = boost::multiprecision::pow(partial_diff, 2) *
44  boost::multiprecision::pow(current_val_error, 2);
45  return static_cast<num_t>(result + generate_error_impl<count + 1, max>(std::forward<F>(f), std::forward<Args>(args)...));
46  }
47 }
