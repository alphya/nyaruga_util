41 // 乱数生成器 Gen を使って乱数値の配列を生成する。

46 //
47 // random_array_impl
48 //
49 // 要素数が指定の数に達した場合、配列を生成して返す。
50 //
51 template<typename T, std::size_t N, typename Random, typename... Args>
52 constexpr typename std::enable_if<
53  sizeof...(Args) == N - 1,
54  sprout::array<T, N>
55 >::type random_array_impl(Random const& rnd, Args const&... args) {
56  return sprout::array<T, N>{{args..., *rnd}};
57 }
58 //
59 // 要素数が指定未満の場合、乱数値を生成して再帰。
60 // Random は、乱数生成器が返す結果型。「生成された乱数値」と「次の状態の乱数生成器」が格納されている。
61 // constexpr では関数型言語のように状態の変更が出来ないため、このような仕様になっている。
62 //
63 template<typename T, std::size_t N, typename Random, typename... Args>
64 constexpr typename std::enable_if<
65  sizeof...(Args) < N - 1,
66  sprout::array<T, N>
67 >::type random_array_impl(Random const& rnd, Args const&... args) {
68  //
69  // operator*() で、「生成された乱数値」を取得。
70  // operator()() で、「次の状態の乱数生成器」を呼び出す。
71  //
72  return random_array_impl<T, N>(rnd(), args..., *rnd);
73 }
74 
75 //
76 // random_array の実装
77 //
78 // 乱数生成器の最初の呼出。
79 //
80 template<typename T, std::size_t N, typename Gen>
81 constexpr sprout::array<T, N> random_array(Gen const& gen) {
82  return random_array_impl<T, N>(gen());
83 }
